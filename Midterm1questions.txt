6) Show what is print out by the following code, if there is an error indicate what is the error.

public class Main{
	public static void main(String[] args){
		rectangle rect = new rectangle(2, 4);
		rect.perimeter();
	}
}

class square{
	int length;

	square(int length){
		this.length = length;
	}

	public int perimeter(){
		return 4*length;
	}
}

class rectangle extend square{
	int length, width;

	rectangle(int length, int width){
		this.length = length;
		this.width = width;
	}

	public int perimeter(){
		return 2*length + 2*width;
	}
}
